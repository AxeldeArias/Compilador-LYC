include number.asm
include macros2.asm

.MODEL LARGE ;Modelo de Memoria
.386 ;Tipo de Procesador
.STACK 200h ;Bytes en el Stack

.DATA
	a1	dd	?
	b1	dd	?
	_1	dd	1.0
	_400	dd	400.0
	_300	dd	300.0
	_2	dd	2.0

	@aux1	dd	?
	@aux2	dd	?


.CODE

start:
	MOV EAX,@DATA
	MOV DS,EAX
	MOV ES,EAX

	FLD _1
	FSTP a1

	FLD _400
	FSTP b1

	FLD b1
	FLD _300
	FCOM
	FSTSW AX
	SAHF
	JBE etiqueta0

	FLD a1
	FLD _1
	FADD
	FSTP @aux1
	FFREE

	FLD @aux1
	FSTP a1

	JMP etiqueta0
	FLD a1
	FLD _2
	FADD
	FSTP @aux2
	FFREE

	FLD @aux2
	FSTP a1

etiqueta0:

	MOV EAX, 4C00h
	INT 21h

	END start
