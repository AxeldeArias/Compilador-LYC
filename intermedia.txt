include number.asm
include macros2.asm

.MODEL LARGE ;Modelo de Memoria
.386 ;Tipo de Procesador
.STACK 200h ;Bytes en el Stack

.DATA
	z	dd	?
	a	dd	?
	b	dd	?
	c	dd	?

	@aux1	dd	?
	@aux2	dd	?


.CODE

start:
	MOV EAX,@DATA
	MOV DS,EAX
	MOV ES,EAX

	GetFloat a

	DisplayFloat a,1
	newline 1

	GetFloat b

	DisplayFloat b,1
	newline 1

	GetFloat c

	DisplayFloat c,1
	newline 1

	FLD a
	FLD b
	FADD
	FSTP @aux1
	FFREE

	FLD @aux1
	FLD c
	FADD
	FSTP @aux2
	FFREE

	FLD @aux2
	FSTP z

	DisplayFloat z,1
	newline 1


	MOV EAX, 4C00h
	INT 21h

	END start
