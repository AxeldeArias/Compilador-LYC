include number.asm
include macros2.asm

.MODEL LARGE ;Modelo de Memoria
.386 ;Tipo de Procesador
.STACK 200h ;Bytes en el Stack

.DATA
	a	dd	?
	b	dd	?
	_1	dd	1.0
	_12	dd	12.0
	_3	dd	3.0
	_6	dd	6.0

	@aux1	dd	?
	@aux2	dd	?


.CODE

start:
	MOV EAX,@DATA
	MOV DS,EAX
	MOV ES,EAX

	FLD _1
	FSTP a

	FLD _12
	FSTP b

	FLD a
	FLD _3
	FMUL
	FSTP @aux1
	FFREE

	FLD @aux1
	FSTP a

	FLD b
	FLD _6
	FXCH
	FDIV
	FSTP @aux2
	FFREE

	FLD @aux2
	FSTP b


	MOV EAX, 4C00h
	INT 21h

	END start
