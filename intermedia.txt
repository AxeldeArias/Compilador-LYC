include number.asm
include macros2.asm

.MODEL LARGE ;Modelo de Memoria
.386 ;Tipo de Procesador
.STACK 200h ;Bytes en el Stack

.DATA
	a	dd	?
	b	dd	?
	c	dd	?
	_2	dd	2.0
	_3	dd	3.0
	_25	dd	25.0
	_1	dd	1.0
	_4	dd	4.0

	@aux1	dd	?
	@aux2	dd	?
	@aux3	dd	?


.CODE

start:
	MOV EAX,@DATA
	MOV DS,EAX
	MOV ES,EAX

	FLD a
	FLD b
	FCOM
	FSTSW AX
	SAHF
	JE etiqueta0

	FLD b
	FLD _2
	FADD
	FSTP @aux1
	FFREE

	FLD @aux1
	FSTP a

	FLD _3
	FLD _25
	FMUL
	FSTP @aux2
	FFREE

	FLD @aux2
	FSTP c

	FLD _1
	FLD _4
	FSUB
	FSTP @aux3
	FFREE

	FLD @aux3
	FSTP b

	etiqueta0

	MOV EAX, 4C00h
	INT 21h

	END start
